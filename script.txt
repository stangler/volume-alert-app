// === 設定パラメータ ===

lookback_period = input.int(20, title="平均出来高の計算期間", minval=5, maxval=100, tooltip="過去何本のローソク足で平均出来高を計算するか")

// === 午前セッション設定 ===

morning_start_hour = input.int(9, title="午前開始時間（時）", minval=0, maxval=23, group="取引時間設定")
morning_start_minute = input.int(0, title="午前開始時間（分）", minval=0, maxval=59, group="取引時間設定")
morning_end_hour = input.int(11, title="午前終了時間（時）", minval=0, maxval=23, group="取引時間設定")

// === 午後セッション設定 ===

afternoon_start_hour = input.int(12, title="午後開始時間（時）", minval=0, maxval=23, group="取引時間設定")
afternoon_start_minute = input.int(30, title="午後開始時間（分）", minval=0, maxval=59, group="取引時間設定")
afternoon_end_hour = input.int(15, title="午後終了時間（時）", minval=0, maxval=23, group="取引時間設定")
afternoon_end_minute = input.int(0, title="午後終了時間（分）", minval=0, maxval=59, group="取引時間設定")

// === 現在の時刻を取得 ===
current_hour = hour(time, "GMT+9")
current_min = minute(time, "GMT+9")

// 指定時間帯内かどうかの判定
morning_start_time_minutes = morning_start_hour * 60 + morning_start_minute
morning_end_time_minutes = morning_end_hour * 60 + morning_end_minute
afternoon_start_time_minutes = afternoon_start_hour * 60 + afternoon_start_minute
afternoon_end_time_minutes = afternoon_end_hour * 60 + afternoon_end_minute
current_time_minutes = current_hour * 60 + current_min

// 午前セッション中かどうか
is_morning_session = enable_morning and (current_time_minutes >= morning_start_time_minutes and current_time_minutes <= morning_end_time_minutes)

// 午後セッション中かどうか
is_afternoon_session = enable_afternoon and (current_time_minutes >= afternoon_start_time_minutes and current_time_minutes <= afternoon_end_time_minutes)

// いずれかのセッション中かどうか
is_within_time_range = is_morning_session or is_afternoon_session

// === 出来高の分析 ===
// 過去の出来高平均を計算
avg_volume = ta.sma(volume[1], lookback_period)

// 現在の出来高が平均の何倍かを計算
volume_ratio = avg_volume > 0 ? volume / avg_volume : 0

// 出来高の上昇・下落判定
volume_increased = volume > volume[1]

// 出来高急増の条件
volume_spike = volume_ratio >= volume_multiplier and is_within_time_range and avg_volume > 0

// === 連続アラートを防ぐためのフィルター ===
var bool alert_triggered = false

// 新しい出来高急増が発生した場合のみアラートを出す
new_spike = volume_spike and not volume_spike[1] and not alert_triggered

// アラートのリセット（出来高が正常レベルに戻った場合）
if not volume_spike
    alert_triggered := false

if new_spike
    alert_triggered := true
